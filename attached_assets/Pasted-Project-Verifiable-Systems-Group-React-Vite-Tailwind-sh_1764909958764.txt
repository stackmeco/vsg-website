  import { useState } from "react";
  import { Link, useLocation } from "wouter";
  import { NAV_ITEMS, NavItem } from "@/config/nav";
  import { cn } from "@/lib/utils";
  // Use existing icon set. If lucide-react is already used elsewhere, you can:
  // import { Menu, X, ChevronDown } from "lucide-react";
  // Otherwise, reuse your existing menu/close/chevron icons.

  const useActivePath = () => {
    const [location] = useLocation();
    return location;
  };

  type DropdownProps = {
    item: NavItem;
    isActive: boolean;
  };

  function DesktopDropdown({ item, isActive }: DropdownProps) {
    if (!item.children?.length) return null;

    return (
      <div className="relative group">
        <button
          type="button"
          aria-haspopup="true"
          aria-expanded="false"
          className={cn(
            "inline-flex items-center gap-1 text-sm font-medium transition-colors",
            isActive
              ? "text-foreground"
              : "text-muted-foreground hover:text-foreground"
          )}
        >
          <span>{item.label}</span>
          {/* Replace with your existing chevron icon */}
          {/* <ChevronDown className="h-3 w-3" /> */}
          <span className="mt-[2px] inline-block h-3 w-3 border-b border-r border-current align-middle [transform:rotate(45deg)]" />
        </button>

        <div className="pointer-events-none absolute left-1/2 top-full z-40 hidden w-[460px] -translate-x-1/2 pt-4 group-hover:pointer-events-auto group-hover:block group-focus-within:pointer-events-auto group-focus-within:block">
          <div className="rounded-lg border border-border bg-background/95 shadow-xl shadow-black/40 backdrop-blur">
            <div className="px-5 py-4 space-y-3">
              {item.children.map((child) => (
                <Link
                  key={child.href}
                  href={child.href}
                  className="flex items-start gap-3 rounded-md px-2 py-2 transition-colors hover:bg-muted/60"
                >
                  <span className="mt-2 h-1.5 w-1.5 shrink-0 rounded-full bg-foreground" />
                  <span className="flex flex-col">
                    <span className="text-sm font-medium text-foreground">
                      {child.label}
                    </span>
                    {child.description && (
                      <span className="text-xs text-muted-foreground leading-snug">
                        {child.description}
                      </span>
                    )}
                  </span>
                </Link>
              ))}
            </div>
          </div>
        </div>
      </div>
    );
  }

  function DesktopNav() {
    const activePath = useActivePath();

    return (
      <nav className="hidden items-center gap-8 md:flex">
        {NAV_ITEMS.map((item) => {
          const isActive =
            item.href &&
            (activePath === item.href ||
              activePath.startsWith(item.href + "/"));

          // CONNECT: reuse existing CTA styling instead of a plain link.
          if (item.key === "connect") {
            // Copy the exact CONNECT button classes or component
            // from your current header implementation and use them here.
            return (
              <Link key={item.key} href={item.href ?? "/connect"}>
                {/* Example structure; replace inner div with your existing CTA */}
                <div className="rounded-sm px-4 py-2 text-xs font-semibold tracking-[0.25em] uppercase bg-orange-500 text-black hover:bg-orange-400 transition-colors">
                  {item.label}
                </div>
              </Link>
            );
          }

          // Simple link (Overview, INSIGHTS) without dropdown
          if (!item.children?.length) {
            return (
              <Link
                key={item.key}
                href={item.href ?? "/"}
                className={cn(
                  "text-sm font-medium transition-colors",
                  item.key === "insights"
                    ? cn(
                        "tracking-[0.25em] uppercase",
                        isActive
                          ? "text-foreground"
                          : "text-muted-foreground hover:text-foreground"
                      )
                    : isActive
                    ? "text-foreground"
                    : "text-muted-foreground hover:text-foreground"
                )}
              >
                {item.label}
              </Link>
            );
          }

          // Dropdown parent (Ventures, Approach, Studio)
          return (
            <DesktopDropdown
              key={item.key}
              item={item}
              isActive={!!isActive}
            />
          );
        })}
      </nav>
    );
  }

  function MobileNav() {
    const [open, setOpen] = useState(false);
    const activePath = useActivePath();

    // Replace these with your existing mobile menu icons if you have them.
    const MenuIcon = () => (
      <span className="flex h-4 w-4 flex-col justify-between">
        <span className="h-[1px] w-full bg-foreground" />
        <span className="h-[1px] w-full bg-foreground" />
        <span className="h-[1px] w-full bg-foreground" />
      </span>
    );
    const CloseIcon = () => (
      <span className="relative h-4 w-4">
        <span className="absolute left-0 top-1/2 h-[1px] w-full -translate-y-1/2 rotate-45 bg-foreground" />
        <span className="absolute left-0 top-1/2 h-[1px] w-full -translate-y-1/2 -rotate-45 bg-foreground" />
      </span>
    );

    return (
      <div className="md:hidden">
        <button
          type="button"
          className="inline-flex items-center justify-center rounded-md border border-border bg-background px-2 py-1.5 text-xs font-medium"
          onClick={() => setOpen((v) => !v)}
          aria-expanded={open}
          aria-label="Toggle navigation"
        >
          {open ? <CloseIcon /> : <MenuIcon />}
        </button>

        {open && (
          <div className="absolute inset-x-0 top-14 z-40 border-b border-border bg-background/95 pb-4 shadow-lg shadow-black/30 backdrop-blur">
            <div className="px-4 pt-3 space-y-4">
              {NAV_ITEMS.map((item) => (
                <div key={item.key} className="space-y-2">
                  <Link
                    href={item.href ?? "/"}
                    className={cn(
                      "block text-sm font-medium",
                      item.key === "insights"
                        ? cn(
                            "tracking-[0.25em] uppercase",
                            activePath === item.href
                              ? "text-foreground"
                              : "text-muted-foreground hover:text-foreground"
                          )
                        : activePath === item.href
                        ? "text-foreground"
                        : "text-muted-foreground hover:text-foreground"
                    )}
                    onClick={() => setOpen(false)}
                  >
                    {item.label}
                  </Link>

                  {item.children?.length && (
                    <div className="space-y-1 border-l border-border/40 pl-3">
                      {item.children.map((child) => (
                        <Link
                          key={child.href}
                          href={child.href}
                          className="block text-xs text-muted-foreground hover:text-foreground"
                          onClick={() => setOpen(false)}
                        >
                          {child.label}
                        </Link>
                      ))}
                    </div>
                  )}
                </div>
              ))}
            </div>
          </div>
        )}
      </div>
    );
  }

  export function GlobalNav() {
    return (
      <header className="sticky top-0 z-50 border-b border-border/40 bg-background/95 backdrop-blur">
        <div className="mx-auto flex h-14 max-w-6xl items-center justify-between px-4 sm:h-16 sm:px-6 lg:px-8">
          {/* LEFT: keep the existing logo block exactly as-is (import and place it here). */}
          <div className="flex items-center gap-3">
            {/* Example: <Logo /> + wordmark */}
          </div>

          {/* RIGHT: desktop nav + search + mobile toggle */}
          <div className="flex items-center gap-4">
            <DesktopNav />
            {/* Keep your existing search icon / command palette trigger here if you have one */}
            <MobileNav />
          </div>
        </div>
      </header>
    );
  }

Key points for premium look:
- Use consistent padding/gap: `gap-8` between top-level items, `px-5 py-4` inside dropdowns.
- Dropdown card: `rounded-lg border border-border bg-background/95 shadow-xl shadow-black/40 backdrop-blur`.
