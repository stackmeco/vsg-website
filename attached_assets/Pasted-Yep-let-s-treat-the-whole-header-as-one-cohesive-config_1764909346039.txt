2) Create a single GlobalNav component (desktop + mobile)
----------------------------------------------------------------

Find the existing header/nav components (look for things like `SiteHeader`, `DesktopNav`, `MobileNav`, `MainNav` in client/src/components). 

Create a new component file:

Implement it roughly as:

  import { Link, useLocation } from "wouter";
  import { NAV_ITEMS, NavItem } from "@/config/nav";
  import { cn } from "@/lib/utils";
  import { Menu, X, ChevronDown } from "lucide-react"; // or existing icon set
  import { useState } from "react";

  // Helper to know active route
  const useActivePath = () => {
    const [location] = useLocation();
    return location;
  };

  type DropdownProps = {
    item: NavItem;
    isActive: boolean;
  };

  function DesktopDropdown({ item, isActive }: DropdownProps) {
    if (!item.children?.length) return null;

    return (
      <div className="relative group">
        <button
          className={cn(
            "inline-flex items-center gap-1 text-sm font-medium transition-colors",
            isActive ? "text-foreground" : "text-muted-foreground hover:text-foreground"
          )}
        >
          <span>{item.label}</span>
          <ChevronDown className="h-3 w-3" />
        </button>

        <div className="pointer-events-none absolute left-1/2 top-full z-40 hidden w-[420px] -translate-x-1/2 pt-4 group-hover:pointer-events-auto group-hover:block">
          <div className="rounded-lg border border-border bg-background/95 shadow-xl backdrop-blur">
            <div className="px-5 py-4 space-y-3">
              {item.children.map((child) => (
                <Link
                  key={child.href}
                  href={child.href}
                  className="flex items-start gap-3 rounded-md px-2 py-2 hover:bg-muted/60"
                >
                  <span className="mt-2 h-1.5 w-1.5 shrink-0 rounded-full bg-foreground" />
                  <span className="flex flex-col">
                    <span className="text-sm font-medium text-foreground">
                      {child.label}
                    </span>
                    {child.description && (
                      <span className="text-xs text-muted-foreground">
                        {child.description}
                      </span>
                    )}
                  </span>
                </Link>
              ))}
            </div>
          </div>
        </div>
      </div>
    );
  }

  function DesktopNav() {
    const activePath = useActivePath();

    return (
      <nav className="hidden items-center gap-8 md:flex">
        {NAV_ITEMS.map((item) => {
          const isActive =
            item.href && (activePath === item.href || activePath.startsWith(item.href + "/"));

          // Dropdown item
          if (item.children?.length) {
            return (
              <DesktopDropdown key={item.key} item={item} isActive={!!isActive} />
            );
          }

          // Simple link item (Overview, Insights, CONNECT)
          return (
            <Link
              key={item.key}
              href={item.href ?? "/"}
              className={cn(
                "text-sm font-medium transition-colors",
                isActive ? "text-foreground" : "text-muted-foreground hover:text-foreground"
              )}
            >
              {item.label}
            </Link>
          );
        })}
      </nav>
    );
  }

  function MobileNav() {
    const [open, setOpen] = useState(false);
    const activePath = useActivePath();

    return (
      <div className="md:hidden">
        <button
          className="inline-flex items-center justify-center rounded-md border border-border bg-background px-2 py-1.5 text-xs font-medium"
          onClick={() => setOpen((v) => !v)}
          aria-expanded={open}
          aria-label="Toggle navigation"
        >
          {open ? <X className="h-4 w-4" /> : <Menu className="h-4 w-4" />}
        </button>

        {open && (
          <div className="absolute inset-x-0 top-14 z-40 border-b border-border bg-background/95 pb-4 shadow-lg backdrop-blur">
            <div className="px-4 pt-3 space-y-4">
              {NAV_ITEMS.map((item) => (
                <div key={item.key} className="space-y-2">
                  <Link
                    href={item.href ?? "/"}
                    className={cn(
                      "block text-sm font-medium",
                      activePath === item.href
                        ? "text-foreground"
                        : "text-muted-foreground hover:text-foreground"
                    )}
                    onClick={() => setOpen(false)}
                  >
                    {item.label}
                  </Link>

                  {item.children?.length && (
                    <div className="space-y-1 border-l border-border/40 pl-3">
                      {item.children.map((child) => (
                        <Link
                          key={child.href}
                          href={child.href}
                          className="block text-xs text-muted-foreground hover:text-foreground"
                          onClick={() => setOpen(false)}
                        >
                          {child.label}
                        </Link>
                      ))}
                    </div>
                  )}
                </div>
              ))}
            </div>
          </div>
        )}
      </div>
    );
  }

  export function GlobalNav() {
    return (
      <header className="sticky top-0 z-50 border-b border-border/40 bg-background/95 backdrop-blur">
        <div className="mx-auto flex h-14 max-w-6xl items-center justify-between px-4 sm:h-16 sm:px-6 lg:px-8">
          {/* Left: logo */}
          <div className="flex items-center gap-3">
            {/* keep existing Logo component or img here */}
          </div>

          {/* Center/right: nav + search + connect button, etc. */}
          <div className="flex items-center gap-4">
            <DesktopNav />
            <MobileNav />
            {/* Keep existing CONNECT button styling and search icon if they’re separate components */}
          </div>
        </div>
      </header>
    );
  }

Adjust imports (Logo component, icons, cn helper, etc.) to match the existing project.
